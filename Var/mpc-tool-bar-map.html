(keymap
 (prev menu-item "prev" mpc-prev :image
       (find-image
	(cond
	 ((not
	   (display-color-p))
	  '((:type pbm :file "mpc/prev.pbm" :foreground "black")
	    (:type xbm :file "mpc/prev.xbm" :foreground "black")
	    (:type xpm :file "low-color/mpc/prev.xpm")
	    (:type xpm :file "mpc/prev.xpm")))
	 ((<
	   (display-color-cells)
	   256)
	  '((:type xpm :file "low-color/mpc/prev.xpm")
	    (:type xpm :file "mpc/prev.xpm")
	    (:type pbm :file "mpc/prev.pbm" :foreground "black")
	    (:type xbm :file "mpc/prev.xbm" :foreground "black")))
	 (t
	  '((:type xpm :file "mpc/prev.xpm")
	    (:type pbm :file "mpc/prev.pbm" :foreground "black")
	    (:type xbm :file "mpc/prev.xbm" :foreground "black")))))
       :enable
       (not
	(equal
	 (cdr
	  (assq 'state mpc-status))
	 "stop"))
       :label "Prev" :vert-only t)
