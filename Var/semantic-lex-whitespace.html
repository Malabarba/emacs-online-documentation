Documentation:
Detect and create whitespace tokens.

((looking-at "\\s-+")
 (if
     (eq
      (semantic-lex-token-class
       (car semantic-lex-token-stream))
      'whitespace)
     (progn
       (setq semantic-lex-end-point
	     (match-end 0))
       (setcdr
	(semantic-lex-token-bounds
	 (car semantic-lex-token-stream))
	semantic-lex-end-point))
